document.addEventListener("DOMContentLoaded", async () => { try { const response = await fetch("https://backend-ufr-ses-i9bo.onrender.com/api/dashboard"); const data = await response.json(); // Affiche les chiffres document.getElementById("total-budget").textContent = data.resume.totalBudget.toLocaleString(); document.getElementById("per-en-attente").textContent = data.resume.perEnAttente; document.getElementById("stocks-critiques").textContent = data.resume.stocksCritiques; document.getElementById("consommation-moyenne").textContent = data.resume.consommationMoyenne; } catch (err) { console.error("Erreur dashboard:", err); } });
